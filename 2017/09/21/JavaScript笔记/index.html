JavaScript笔记

1.考察this

    var length = 10;
    function fn(){
    console.log(this.length);
    }

    var obj = {
    length:5,method:function(fn){fn();;arguments[0]();}
    obj.method(fn,1);

取对象可以用点操作符还可以用中括号，第二次执行时相当于调用了arguments方法，this指向了arguments,传了两个参数，故arguments长度为2

2.var和函数的提前声明

    function fn(a){
        console.log(a);
        var a=2;       //var声明
        function a(){  //函数声明
            console.log(a);
        }
    }
    fn(1);

3.局部变量和全局变量

    var f =true;
    if(f===true){
        var a = 10;//全局变量
    }
    function fn(){
        b=20;//全局变量(没有用var声明的变量)
        var c=30;//局部变量
    }
    fn();
    console.log(a);
    console.log(b);
    console.log(c);

function内的新声明的变量才是局部变量，而没有用var声明的变量在哪里都是全局变量。再次提醒切记只有function(){}内新声明的才能是局部变量，while{…}、if{…}、for(..) 之内的都是全局变量（除非本身包含在function内）

4.变量隐式声明

    if('a'in window){//a已提前声明好了，'a'in window为tru， a被赋值
        var a = 10;
    }
    console.log(a);

function和var会提前声明，function{}内变量也会提前声明

5.给基本类型数据添加属性，不报错，但取值时是undefined

    var a =10;
    a.pro = 10;//a.pro返回值是undefined
    console.log(a.pro + a);//a.pro + a =undefined + 10 返回的是NaN
    var s ="hello";
    s.pro = "world";//s.pro返回值是undefined
    console.log(s.pro + s);//undefined+hello 结果转换成字符串

6.函数声明优于变量声明

    console.log(typeof fn);
    function fn(){};//声明fn函数
    var fn;//声明fn变量
    f();

在代码逐行执行前，函数声明和变量声明会提前进行，而函数声明又会优于变量声明，这里的优于可以理解为晚于变量声明后，如果函数名和变量名相同，函数声明就能覆盖变量声明
